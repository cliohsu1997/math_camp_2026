\section{Day 1 --- Foundations (Sections 1.1--1.9)}
%==============================================================================

\textit{Approximate duration: 3 hours. See Day~1 slides (Sections 1.1--1.9).}

\subsection{1.1 --- Notation and Number Systems}
\begin{itemize}[leftmargin=*]
    \item Number systems: $\mathbb{N}$ (counting), $\mathbb{Z}$, $\mathbb{Q}$, $\mathbb{R}$, $\mathbb{R}^n$;
      nesting $\mathbb{N}\subset\mathbb{Z}\subset\mathbb{Q}\subset\mathbb{R}$.
    \item Symbols: $\in$, $\subset$, $\forall$, $\exists$, $\Rightarrow$, $\Leftrightarrow$.
    \item Scalars versus vectors: $\mathbf{x}=(x_1,\ldots,x_n)\in\mathbb{R}^n$;
      dot product $\mathbf{p}\cdot\mathbf{x}=\sum_i p_i x_i$.
    \item Practice: identify the smallest set containing a given object
      (e.g.\ counts in $\mathbb{N}$; prices in $\mathbb{Q}$; order vectors in product sets).
\end{itemize}

\subsection{1.2 --- Logic and Mathematical Statements}
\begin{itemize}[leftmargin=*]
    \item $P$ \textbf{only if} $Q$ ($P\Rightarrow Q$): $Q$ necessary for $P$.
    \item $P$ \textbf{if} $Q$ ($Q\Rightarrow P$): $Q$ sufficient for $P$.
    \item $P$ \textbf{iff} $Q$ ($P\Leftrightarrow Q$): necessary and sufficient.
    \item Contrapositive: $\neg Q\Rightarrow\neg P$ equivalent to $P\Rightarrow Q$.
    \item Running example: IKEA budget set
      $\mathcal{B}=\{\mathbf{x}\in\mathbb{N}^3:\mathbf{p}\cdot\mathbf{x}\le B\}$;
      practice with $\in$, $\subset$, $\forall$, $\exists$, $\Rightarrow$, and contrapositive.
\end{itemize}

\subsection{1.3 --- Sets and Set Operations}
\begin{itemize}[leftmargin=*]
    \item Notation: $\in$, $\notin$, $\subset$, $=$, $\emptyset$, roster and set-builder forms.
    \item Union, intersection, complement, difference; Cartesian product $A\times B$.
    \item Economic sets: budget sets, consumption sets; product spaces such as
      $\mathbb{Q}\times\mathbb{N}$ versus constraint sets like $\mathcal{B}$.
    \item Practice: describe $\mathcal{B}\cap\mathcal{H}$, $\mathcal{B}\cup\mathcal{H}$,
      $\mathcal{B}\setminus\mathcal{H}$; distinguish elements from subsets.
\end{itemize}

\subsection{1.4 --- Functions}
\begin{itemize}[leftmargin=*]
    \item Function $f\colon A\to B$: domain, codomain, range $f(A)\subseteq B$.
    \item Injective (one-to-one) and surjective (onto); codomain versus range.
    \item Operations: $(f\pm g)$, $(cf)$, $(fg)$, composition $(f\circ g)(x)=f(g(x))$.
    \item A rule must assign \textbf{exactly one} output per input (not a correspondence).
    \item Practice: domain, codomain, range, injectivity, and surjectivity
      (e.g.\ partner age as a function of your age).
\end{itemize}

\subsection{1.5 --- Supremum and Infimum}
\begin{itemize}[leftmargin=*]
    \item Example first: reservation price / willingness to pay when the purchase set
      $P=\{p\in\mathbb{R}_+:p<\bar p\}$ has no maximum but $\sup P=\bar p$.
    \item $\max S$ / $\min S$ versus $\sup S$ / $\inf S$ (least upper bound, greatest lower bound).
    \item When extrema are attained versus only approached (e.g.\ $S=[0,1)$, $\sup S=1\notin S$).
    \item If $\max S$ exists, $\max S=\sup S$; always $\inf S\le x\le\sup S$ for $x\in S$.
\end{itemize}

\subsection{1.6 --- Algebra Revision}
\begin{itemize}[leftmargin=*]
    \item Taylor expansion around $a$:
      $f(a+h)=\sum_{k=0}^{\infty}\frac{f^{(k)}(a)}{k!}h^k$ (first-order: $f(a+h)\approx f(a)+f'(a)h$).
    \item Log and exponential rules; series $\ln(1+x)$ and $e^x$.
    \item \textbf{Why logs:} $\Delta\ln x\approx \Delta x/x$ tracks percentage changes;
      elasticity $\varepsilon_{y,x}=\partial\ln y/\partial\ln x$; log--log regression.
    \item Cobb--Douglas: $U=x_1^{\alpha}x_2^{1-\alpha}$, $Y=AK^{\alpha}L^{1-\alpha}$;
      linear in logs; constant elasticities and expenditure shares.
    \item Local log approximation of any smooth $u>0$ near $\bar{\mathbf{x}}$;
      Cobb--Douglas as a modeling choice when used globally.
    \item Homogeneity of degree~$k$: $f(t\mathbf{x})=t^k f(\mathbf{x})$;
      Cobb--Douglas utility and production are degree~$1$ (CRS: double inputs $\Rightarrow$ double output).
\end{itemize}

\subsection{1.7 --- Single-Variable Derivatives}
\begin{itemize}[leftmargin=*]
    \item Formal definition: $f'(a)=\lim_{h\to 0}\frac{f(a+h)-f(a)}{h}$;
      meaning as slope and marginal rate ($MC$, $MU$, $MP_K$).
    \item Rules: constants, sums, power, product, quotient, chain rule;
      $(e^x)'=e^x$, $(\ln x)'=1/x$, $(a^x)'=a^x\ln a$; inverse-function rule (stated).
    \item Practice: cubic cost, monopolist profit, Cobb--Douglas $Y(K)$, composite cost (chain rule).
    \item \textbf{Euler's theorem} (for $C^1$ homogeneous $f$ of degree~$k$):
      $\sum_i x_i f_{x_i}=k f(\mathbf{x})$ (proof via $t\mapsto f(t\mathbf{x})$).
    \item Adding-up problem: under CRS and competitive factor markets,
      $PQ=rK+wL$; IRS ($k>1$) and DRS ($k<1$) and payment exhaustion.
\end{itemize}

\subsection{1.8 --- Multivariate Calculus}
\begin{itemize}[leftmargin=*]
    \item Level curves $\mathcal{L}_c=\{(x,y):f(x,y)=c\}$ and contour maps;
      3D slice picture for $z=f(x,y)$.
    \item Roadmap: partials $\Rightarrow$ total differential $\Rightarrow$ directional derivative
      $\Rightarrow$ cross-partials (Schwarz).
    \item Partial derivatives (limit definition; fix one variable);
      $MP_K=\partial Y/\partial K$ with $L$ fixed.
    \item Total differential: add/subtract trick, MVT on each bracket,
      $dz=f_x\,dx+f_y\,dy$; Cobb--Douglas output example $dY=MP_K\,dK+MP_L\,dL$.
    \item Gradient $\nabla f$: perpendicular to level curves; points uphill;
      $D_{\mathbf{u}}f=\nabla f\cdot\mathbf{u}=\|\nabla f\|\cos\theta$.
    \item Directional derivative along unit direction $\mathbf{u}$;
      example: subsidy requiring input changes in ratio $2{:}1$.
    \item Cross-partials and Schwarz's theorem (stated); $Y_{KL}$ and complements versus substitutes.
\end{itemize}

\subsection{1.9 --- Limits}
\begin{itemize}[leftmargin=*]
    \item Motivation: Berkeley's critique of ``infinitely small'' $h$ in early calculus.
    \item Sequence limit ($\varepsilon$--$N$): quantifier order and convergence.
    \item Function limit ($\varepsilon$--$\delta$): $0<|x-a|<\delta\Rightarrow|f(x)-L|<\varepsilon$;
      example $f(x)=x+1$ for $x\neq 1$, $f(1)=3$, $\lim_{x\to 1}f(x)=2\neq f(1)$.
    \item Continuity at $a$: $\lim_{x\to a}f(x)=f(a)$;
      differentiability at $a$: derivative limit exists.
    \item Differentiable $\Rightarrow$ continuous; converse fails ($|x|$ at $0$; Weierstrass function).
    \item Practice: $\varepsilon$--$\delta$ proofs for $f'(a)=2a$ when $f(x)=x^2$ and for
      $\lim_{x\to 1}f(x)=2$.
\end{itemize}

\subsection{Appendix (Slides)}
\begin{itemize}[leftmargin=*]
    \item Derivations of single-variable rules: constants, sums, powers, product, quotient.
    \item Key limits for $(e^x)'$ and $(\ln x)'$; exponential and logarithm derivative proofs.
\end{itemize}

\section{Day 2 --- More Calculus and Real Analysis Bridge}
%==============================================================================

\textit{Approximate duration: 3 hours. See Day~2 slides (Sections 2.1--2.6).}

\subsection{2.1 --- Implicit Function Theorem}
\begin{itemize}[leftmargin=*]
    \item Explicit versus implicit relationships; when $F(x,y)=0$ cannot be solved in closed form.
    \item Comparative statics: $F(x,\theta)=0$ defines $y^*(\theta)$; goal is $\dfrac{dy^*}{d\theta}$.
    \item \textbf{Core recipe:} totally differentiate $F(x,y)=0$ to get
      $\dfrac{dy}{dx}=-\dfrac{F_x}{F_y}$ when $F_y\neq 0$.
    \item Regularity: $F_y\neq 0$ means finite slope locally; vertical tangent when $F_y=0$.
    \item \textbf{Practice (Together):} nonlinear demand $D(p,\alpha)=\alpha e^{-p}$, supply $S(p)=p$;
      no closed form for $p^*(\alpha)$, but IFT still gives $\dfrac{dp^*}{d\alpha}$.
    \item \textbf{Applications:} market clearing $E(p,\alpha)=D(p,\alpha)-S(p)=0$;
      linear example checks IFT against explicit $p^*(\alpha)=\dfrac{\alpha-c}{b+d}$.
    \item Direct versus indirect effects; chain-rule derivation of $\dfrac{dp^*}{d\alpha}=-\dfrac{E_\alpha}{E_p}$.
    \item \textbf{Tax incidence:} per-unit tax on sellers; $\dfrac{dp^*}{dt}=\dfrac{d}{b+d}$;
      tax borne more heavily by the \textbf{less elastic} side.
    \item \textbf{Appendix (for interest):} several-variable / Jacobian form of IFT.
\end{itemize}

\subsection{2.2 --- EVT, MVT, and Existence}
\begin{itemize}[leftmargin=*]
    \item \textbf{Open versus closed sets} (budget, strict budget, reservation price examples).
    \item EVT on closed interval $[a,b]$: continuous $f$ attains max and min.
    \item When EVT fails: not closed, not bounded, or not continuous (failure figure).
    \item \textbf{Compact} in $\mathbb{R}^n$ (camp level): closed and bounded (Heine--Borel, stated).
    \item \textbf{Weierstrass EVT:} continuous $f$ on compact $K$ attains max and min.
    \item \textbf{Bridge:} compact feasible set $+$ continuous objective $\Rightarrow$ optimum exists.
    \item Rolle's theorem; MVT; labor/output economics example.
    \item \textbf{Appendix (for interest):} Weierstrass EVT proof idea.
\end{itemize}

\subsection{2.3 --- Integration}
\begin{itemize}[leftmargin=*]
    \item Riemann integral as signed area / accumulation; definite versus indefinite integrals.
    \item \textbf{For interest:} Lebesgue integral (measures level sets; handles sparse jumps).
    \item Fundamental Theorem of Calculus: $F'(x)=f(x)$ $\Rightarrow$
      $\displaystyle\int_a^b f(x)\,dx=F(b)-F(a)$.
    \item Integration by parts: $\displaystyle\int u\,dv=uv-\int v\,du$ (product-rule proof);
      example $\displaystyle\int x e^x\,dx$.
    \item \textbf{Probability link:} PMF and CMF (discrete dice); PDF and CDF (continuous Unif$(0,1)$);
      FTC link $F'(x)=f(x)$.
    \item Expectation for non-negative continuous $X$:
      $\mathbb{E}[X]=\displaystyle\int_0^\infty x f(x)\,dx=\int_0^\infty(1-F(x))\,dx$.
    \item \textbf{Appendix (for interest):} expectation-from-CDF proof (integration by parts).
\end{itemize}

\subsection{2.4 --- Leibniz Rule and Fubini}
\begin{itemize}[leftmargin=*]
    \item \textbf{Fixed limits:} if $F(\theta)=\displaystyle\int_a^b f(x,\theta)\,dx$, then under regularity
      $\dfrac{dF}{d\theta}=\displaystyle\int_a^b \dfrac{\partial f}{\partial\theta}(x,\theta)\,dx$.
    \item Regularity (camp level): $\partial f/\partial\theta$ exists and $f$, $\partial f/\partial\theta$
      are continuous on $[a,b]\times I$.
    \item \textbf{Variable limits:} when $a(\theta)$, $b(\theta)$ move,
      $F'(\theta)$ adds endpoint terms plus the integral of $\partial f/\partial\theta$.
    \item When fixed-limit Leibniz fails: moving discontinuity (step function) example.
    \item \textbf{Fubini:} if $\displaystyle\iint_D |f(x,y)|\,dx\,dy<\infty$, then
      $\displaystyle\iint_D f=\int\!\int f$ (order of integration can be swapped).
    \item Camp shortcut: bounded continuous $f$ on a nice region; Tonelli (for interest) when $f\ge 0$.
    \item Example: $\displaystyle\iint_D (x+y)\,dx\,dy$ on triangle $D=\{0\le x\le 1,\ 0\le y\le x\}$.
\end{itemize}

\subsection{2.5 --- Taylor Expansion and Linearization}
\begin{itemize}[leftmargin=*]
    \item One-variable Taylor through order $n$; multivariable $C^2$ form with $\nabla f$ and Hessian $H_f$.
    \item First-order truncation: $f(\mathbf{a}+\mathbf{h})\approx f(\mathbf{a})+\nabla f(\mathbf{a})\cdot\mathbf{h}$.
    \item Second-order truncation: adds $\tfrac12\,\mathbf{h}^{\mathsf T}H_f(\mathbf{a})\,\mathbf{h}$.
    \item Two-variable explicit formula from matrix multiplication (optional if Hessian is familiar).
    \item When linearization works: small shifts, modest curvature, differentiability;
      fails with large shifts, kinks, strong nonlinearity.
    \item \textbf{Practice (Together):} first-order two-variable Taylor for $f(x,y)=x^2+y$ at $(1,1)$.
    \item \textbf{Take home:} second-order two-variable Taylor (e.g.\ $f(x,y)=\sin x+\cos y$ at $(0,0)$).
    \item \textbf{Appendix (for interest):} one-variable Taylor proof; It\^o's lemma and Brownian motion.
\end{itemize}

\subsection{2.6 --- Uniform Convergence}
\begin{itemize}[leftmargin=*]
    \item Pointwise convergence: $f_n(x)\to f(x)$ for each fixed $x$.
    \item Uniform convergence: one $N$ works for all $x$ in the domain.
    \item Example: $f_n(x)=x^n$ on $[0,1]$ converges pointwise but not uniformly.
    \item \textbf{Why it matters:} uniform convergence justifies swapping limits with continuity,
      integration, and (with extra conditions) differentiation.
    \item Link to section~2.4: Leibniz/Fubini need regularity so limits and integrals can be interchanged safely.
\end{itemize}

\subsection{Appendix (Day 2 Slides)}
\begin{itemize}[leftmargin=*]
    \item IFT --- several variables (Jacobian form).
    \item Weierstrass EVT --- proof idea.
    \item Expectation from the CDF (integration by parts).
    \item Leibniz rule failure (moving step).
    \item Taylor expansion proof (one variable); It\^o's lemma and Brownian motion (for interest).
\end{itemize}

\section{Day 4 --- Optimization}
%==============================================================================

\textit{Approximate duration: 3 hours.
Convex \textbf{sets} (feasible regions) are covered on Day~3 --- callback only.
Focus: FOC/SOC, concave/quasiconcave \textbf{functions}, Lagrange, envelope, geometric Farkas $\Rightarrow$ KKT.}

\subsection{4.1 --- Unconstrained Optimization (1D, then multivariable)}
\begin{itemize}[leftmargin=*]
    \item \textbf{Single variable:} local versus global optima; interior FOC $f'(x^*)=0$.
    \item \textbf{SOC via concavity:} concave $\Rightarrow$ first-order Taylor above $f$; at $f'(x^*)=0$, local max.
      Convex dual for min. Figure: tangent above/below graph.
    \item \textbf{$f''\le 0\Leftrightarrow$ concave} ($C^2$): Taylor proof for $\Rightarrow$;
      symmetric second-difference limit for $\Leftarrow$.
    \item Remark: $f(x)=-x^4$ strictly concave but $f''(0)=0$; $f''<0$ sufficient, not necessary, for strict concavity.
    \item \textbf{Multivariable:} $\nabla f(\mathbf{x}^*)=\mathbf{0}$; SOC stated via concavity / $H_f$ ND;
      proof in appendix.
    \item \textbf{Together:} monopoly $\pi(q)=pq-q^2$. \textbf{Take home:} two-input convex cost $C=q_1^2+q_2^2$.
\end{itemize}

\subsection{4.2 --- Concavity, Quasiconcavity, and Global Optima}
\begin{itemize}[leftmargin=*]
    \item \textbf{Convex/concave functions} (not convex sets --- see Day~3):
      secant below graph (concave) / above graph (convex); upper and lower contour sets.
    \item \textbf{Quasiconcave} (maximization): along $\mathbf{z}=\lambda\mathbf{x}+(1-\lambda)\mathbf{y}$,
      $f(\mathbf{z})\ge\min\{f(\mathbf{x}),f(\mathbf{y})\}$; equivalently upper contour sets are convex.
    \item \textbf{Prove equivalence} quasiconcavity $\Leftrightarrow$ convex upper contour sets (two-direction proof).
    \item Concave $\Rightarrow$ quasiconcave; Cobb--Douglas utility is quasiconcave (stated).
    \item \textbf{Global sufficiency:} quasiconcave $f$ on convex feasible set $\mathcal{B}$
      (Day~3 callback) $\Rightarrow$ interior FOC/Lagrange solution is a global maximum.
    \item \textbf{Together:} Cobb--Douglas quasiconcavity. \textbf{Take home:} $u=x_1x_2$.
\end{itemize}

\subsection{4.3 --- Equality Constraints and Lagrange Multipliers}
\begin{itemize}[leftmargin=*]
    \item Problem: $\max f(x,y)$ subject to $h(x,y)=0$; constraint defines a \textbf{curve} in the plane.
    \item \textbf{3D picture:} surface $z=f(x,y)$; constraint $h(x,y)=0$ is a curve on the domain;
      optimum on the curve.
    \item \textbf{Tangent argument:} any local direction $\mathbf{d}$ along the curve satisfies
      $\nabla h\cdot\mathbf{d}=0$; if $\nabla f\cdot\mathbf{d}\neq 0$, then $f$ increases along the curve
      locally $\Rightarrow$ not optimal. Hence $\nabla f\parallel\nabla h$.
    \item Lagrangian (equality): $\mathcal{L}=f+\mu h$; FOC $\nabla f+\mu\nabla h=0$, $h=0$.
    \item $\mu$ \textbf{unrestricted} (either sign): equality allows both tangent directions.
    \item Shadow price interpretation; Cobb--Douglas budget (callback Day~1).
    \item \textbf{Together:} budget max with $\mathcal{L}=u+\mu(w-p\cdot\mathbf{x})$.
      \textbf{Take home:} cost min with production equality.
\end{itemize}

\subsection{4.4 --- Envelope Theorem}
\begin{itemize}[leftmargin=*]
    \item \textbf{Placement:} after Lagrange setup, before inequalities/KKT.
    \item Value function $V(\theta)=\max_{\mathbf{x}} f(\mathbf{x};\theta)$ (subject to constraints).
    \item \textbf{Unconstrained:}
      $\dfrac{dV}{d\theta}=\dfrac{\partial f}{\partial\theta}\Big|_{\mathbf{x}^*(\theta)}$
      --- no need to differentiate $\mathbf{x}^*(\theta)$.
    \item \textbf{Constrained:}
      $\dfrac{dV}{d\theta}=\dfrac{\partial \mathcal{L}}{\partial\theta}\Big|_{(\mathbf{x}^*,\boldsymbol{\mu}^*,\boldsymbol{\lambda}^*)}$.
    \item Economic examples: profit vs.\ price shifter; indirect utility vs.\ income (one each).
    \item Contrast with chain rule through $\mathbf{x}^*(\theta)$: envelope drops terms that vanish at the optimum.
    \item \textbf{Together:} $\pi^*(p)$ envelope gives $d\pi^*/dp=q^*$.
      \textbf{Take home:} $V(p,w)$, $\partial V/\partial w=\lambda$.
\end{itemize}

\subsection{4.5 --- Inequalities, Geometric Farkas, and KKT}
\begin{itemize}[leftmargin=*]
    \item Inequalities $g_j(\mathbf{x})\le 0$; binding versus slack; complementary slackness
      $\lambda_j g_j(\mathbf{x}^*)=0$.
    \item \textbf{Feasible directions} at a boundary point form a \textbf{convex cone}:
      $\nabla g_j\cdot\mathbf{d}\le 0$ for active constraints.
    \item No improving feasible direction: for a max, no $\mathbf{d}$ with
      $\nabla f\cdot\mathbf{d}>0$ and $\nabla g_j\cdot\mathbf{d}\le 0$.
    \item \textbf{Geometric Farkas (picture):} in the $(x_1,x_2)$ plane, active $\nabla g_j$ span cone $K$;
      at optimum $-\nabla f\in K$. Figure: \texttt{farkas\_cone\_2d.png} (Day~4 slides).
    \item \textbf{KKT} (stated): stationarity, primal/dual feasibility, complementary slackness.
    \item \textbf{Sign mnemonic} --- Convention A only on slides: $\mathcal{L}=f+\sum_j\lambda_j g_j+\sum_k\mu_k h_k$
      (all plus); see table below.
    \item Nonnegativity $x_i\ge 0$ and corner solutions (conceptual).
    \item \textbf{Sufficiency (brief):} quasiconcave $f$ + convex $\mathcal{B}$ $\Rightarrow$ KKT point is global;
      constraint qualification (e.g.\ LICQ) for necessity (name only).
    \item \textbf{Together:} budget + inequality KKT; min cost with $\ge$ constraint.
      \textbf{Take home:} corner $x_1=0$.
    \item \textbf{Appendix (for interest):} algebraic Farkas proof; link to \texttt{duality\_notes}.
\end{itemize}

\subsection{KKT sign mnemonic (reference)}

\noindent\textbf{Convention A (fixed form).}\ Use $\mathcal{L}=f+\sum_j\lambda_j g_j$ with the constraint
written exactly as in the problem ($g\le 0$ or $g\ge 0$):

\begin{center}
\small
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Problem} & \textbf{Constraint} & $\boldsymbol{\lambda}$ \\
\midrule
$\max f$ & $g\le 0$ & $\lambda\ge 0$ \\
$\max f$ & $g\ge 0$ & $\boldsymbol{\lambda\le 0}$ \\
$\min f$ & $g\le 0$ & $\lambda\ge 0$ \\
$\min f$ & $g\ge 0$ & $\boldsymbol{\lambda\le 0}$ \\
\midrule
\multicolumn{3}{l}{Equality $h=0$: $\mathcal{L}=f+\mu h$; $\mu$ \textbf{unrestricted}.} \\
\bottomrule
\end{tabular}
\end{center}

\noindent\textbf{Short rule (Convention A):}\ $\le$ on the constraint $\Rightarrow$ $\lambda\ge 0$;
$\ge$ on the constraint $\Rightarrow$ $\lambda\le 0$.

\medskip
\noindent\textbf{Memory tricks (sign of objective $\times$ sign of constraint inequality):}
\begin{itemize}[nosep,leftmargin=*]
  \item $\max f$ $(+)$ with $g\le 0$ $(-)$:\ ``opposite'' $\Rightarrow$ $\lambda\ge 0$.
  \item $\min f$ $(-)$ with $g\le 0$ $(-)$:\ ``$- \times -$'' $\Rightarrow$ $\lambda\ge 0$.
  \item $\max f$ $(+)$ with $g\ge 0$ $(+)$:\ ``same'' $\Rightarrow$ $\boldsymbol{\lambda\le 0}$.
  \item $\min f$ $(-)$ with $g\ge 0$ $(+)$:\ ``opposite'' $\Rightarrow$ $\boldsymbol{\lambda\le 0}$.
\end{itemize}

\medskip
\noindent\textbf{Convention B (footnote only --- not used on Day~4 slides).}\ An alternative is to flip
the constraint into $\mathcal{L}$ so $\lambda_j\ge 0$ always; see fall-course notes if needed.

\begin{itemize}[leftmargin=*]
    \item Multivariate SOC proof (Hessian ND).
    \item Algebraic Farkas lemma and KKT derivation.
    \item Bordered Hessian / SOC for constrained problems (for interest only).
    \item Take-home solutions (§4.1--4.5).
\end{itemize}

\section{Day 5 --- Dynamic Programming}
%==============================================================================

\textit{Approximate duration: 3 hours.}

\subsection{Motivation and Framework}
\begin{itemize}[leftmargin=*]
    \item Sequential decision-making over time.
    \item Intertemporal trade-offs.
