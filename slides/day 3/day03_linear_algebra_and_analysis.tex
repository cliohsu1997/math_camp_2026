\section{3.1 --- Fixed Point Theorems}

\begin{frame}{3.1 Fixed points --- equilibrium as a solution}
  \footnotesize
  \textbf{Fixed point.}\ A point $x^*$ is a \textbf{fixed point} of a map $T$ if $T(x^*)=x^*$.

  \medskip
  \textbf{Economics reading.}\ Many equilibria are fixed-point problems:
  \begin{itemize}
    \item \textbf{Market clearing:}\ a price $p^*$ with excess demand $E(p^*)=0$.
    \item \textbf{Game theory:}\ a strategy profile that is a best response to itself (Nash).
  \end{itemize}

  \medskip
  \textbf{Two useful existence tools (today).}
  \begin{itemize}
    \item \textbf{Contraction Mapping Theorem:}\ existence and uniqueness of a fixed point; iteration converges.
    \item \textbf{Brouwer:}\ continuous map on a closed interval / convex set.
  \end{itemize}
\end{frame}

\begin{frame}{3.1 Contraction Mapping Theorem}
  \footnotesize
  \textbf{Contraction.}\ A map $T$ \textbf{shrinks distances} if there is $k\in(0,1)$ such that
  $|T(x)-T(y)|\le k|x-y|$ for all $x,y$ in the domain.

  \medskip
  \textbf{Contraction Mapping Theorem.}\ If $T\colon[a,b]\to[a,b]$ is a contraction, then
  \begin{itemize}
    \item \textbf{Existence:}\ there is a fixed point $x^*$ with $T(x^*)=x^*$.
    \item \textbf{Uniqueness:}\ that fixed point is \textbf{unique}.
    \item \textbf{Iteration:}\ for any starting value $x_0\in[a,b]$, the sequence
      $x_n=T(x_{n-1})$ converges to $x^*$.
  \end{itemize}
\end{frame}

\begin{frame}{3.1 Why iteration converges}
  \footnotesize
  \textbf{Step 1 --- Start anywhere.}\
  Pick any $x_0$. Define $x_n=T(x_{n-1})$ for $n=1,2,\ldots$

  \medskip
  \textbf{Step 2 --- Jumps shrink exponentially.}\
  Each new step is at most $k$ times the previous jump:
  \[
    |x_{n+1}-x_n|=|T(x_n)-T(x_{n-1})|\le k|x_n-x_{n-1}|\le k^n|x_1-x_0|.
  \]
  Since $k<1$, the moves get \textbf{exponentially smaller} --- the path barely changes after a while.

  \medskip
  \textbf{Step 3 --- The sequence converges.}\
  Step~2 implies $x_n\to x^*$ for some limit $x^*$
  (check it is a \textbf{Cauchy sequence} for a rigorous proof).
  So iteration from \textbf{any} starting point drives $x_n$ toward the same $x^*$.

  \medskip
  \textbf{Step 4 --- The limit is a fixed point.}\
  $T$ is continuous, so taking limits in $x_n=T(x_{n-1})$ gives
  $x^*=\lim x_n=\lim T(x_{n-1})=T(\lim x_{n-1})=T(x^*)$.
\end{frame}

\begin{frame}{3.1 Walrasian price adjustment}
  \footnotesize
  \textbf{Market clearing.}\ A price $p^*$ \textbf{clears the market} when excess demand is zero:
  $E(p^*)=0$.
  Here $E(p)$ can take \textbf{any form} (from supply and demand behind the scenes).

  \medskip
  \textbf{Price adjustment (nonlinear in $E$).}\ Update the price each period:
  \[
    p_{t+1}=T(p_t)=p_t+\lambda\bigl(1-e^{-\alpha E(p_t)}\bigr),
    \qquad \lambda=\tfrac{1}{5},\ \alpha=0.1.
  \]
  When $|E|$ is small, $1-e^{-\alpha E}\approx \alpha E$ --- move a fraction of excess demand toward balance.
  Clearing satisfies $E(p^*)=0\ \Leftrightarrow\ T(p^*)=p^*$.
  If $T$ is a \textbf{contraction} on a price range, then $p_t\to p^*$ from any starting $p_0$.
\end{frame}

\begin{frame}{3.1 Brouwer --- one dimension}
  \footnotesize
  \textbf{Brouwer (one dimension).}\ If $T\colon[a,b]\to[a,b]$ is \textbf{continuous}, then
  $T$ has \textbf{at least one} fixed point $x^*$ with $T(x^*)=x^*$.

  \medskip
  \textbf{Picture.}\ Plot $y=T(x)$ against $y=x$. Any crossing is a fixed point.
  Continuity forces the graph to meet the diagonal at least once on $[a,b]$.
\end{frame}

\begin{frame}{3.1 Brouwer --- general statement}
  \footnotesize
  \textbf{Brouwer (several variables).}\ Let $K\subset\mathbb{R}^n$ be \textbf{nonempty},
  \textbf{compact}, and \textbf{convex}. If $T\colon K\to K$ is \textbf{continuous}, then
  $T$ has at least one fixed point $x^*\in K$ with $T(x^*)=x^*$.

  \medskip
  \textbf{Camp vocabulary.}
  \begin{itemize}
    \item \textbf{Compact} in $\mathbb{R}^n$: closed and bounded (Heine--Borel).
    \item \textbf{Convex}: the segment between any two points in $K$ stays in $K$.
  \end{itemize}

  \medskip
  \textbf{One dimension.}\ $K=[a,b]$ is compact and convex, so the 1D case is a special case.
\end{frame}

\begin{frame}{3.1 Brouwer --- Cournot duopoly}
  \footnotesize
  \textbf{Two firms.}\ Firm $i$ chooses $q_i\in[0,\bar q]$.
  \textbf{Nonlinear} inverse demand $p=a-(q_1+q_2)^2$ (e.g.\ $a=8$, $\bar q=2$).

  \medskip
  \textbf{Profit and FOC.}\ $\pi_i=(a-(q_1+q_2)^2)q_i$. Firm~1's best response solves
  \[
    a-3q_1^2-4q_1 q_2-q_2^2=0
    \qquad\text{(similarly for firm~2 with $q_1,q_2$ swapped).}
  \]
  The map $q_j\mapsto BR_i(q_j)$ is \textbf{nonlinear} --- no formula like
  ``$(a-c-q_j)/2$''.

  \medskip
  \textbf{Fixed point = Nash.}\ $T(q_1,q_2)=\bigl(BR_1(q_2),BR_2(q_1)\bigr)$ is continuous on
  $K=[0,\bar q]^2$. \textbf{Brouwer} guarantees an equilibrium, but you generally solve
  $T(q_1^*,q_2^*)=(q_1^*,q_2^*)$ by \textbf{algebra or numerics}
  (here one symmetric solution is $q_1^*=q_2^*=\sqrt{a/8}$).
\end{frame}

\section{3.2 --- Separation Theorem}

\begin{frame}{3.2 Separating Hyperplane Theorem}
  \footnotesize
  \textbf{Setup.}\ Two nonempty \textbf{convex} sets $A,B\subset\mathbb{R}^n$ with $A\cap B=\varnothing$.

  \medskip
  \textbf{Theorem.}\ If \textbf{at least one} of $A,B$ is \textbf{compact}, then there exist
  $\mathbf{p}\neq\mathbf{0}$ and $\beta\in\mathbb{R}$ such that
  $\mathbf{p}\cdot a\le\beta\le\mathbf{p}\cdot b$ for all $a\in A$, $b\in B$.
  Geometrically: the hyperplane $\mathbf{p}\cdot x=\beta$ separates $A$ from $B$.
  (Both need not be compact --- if only $B$ is compact, swap labels.)

  \medskip
  \textbf{In two goods ($x=(x_1,x_2)$).}\ The line $p_1 x_1+p_2 x_2=\beta$ puts all of $A$ on one side
  ($\mathbf{p}\cdot a\le\beta$) and all of $B$ on the other ($\mathbf{p}\cdot b\ge\beta$);
  $\mathbf{p}$ is the normal, $\beta$ fixes the line's position.

  \medskip
  \textbf{Remark (single point).}\ $B=\{b_0\}$ is compact. If $b_0\notin A$ and $A$ is closed and convex,
  a separating hyperplane exists even when $A$ is not compact.

  \medskip
  \textbf{Remark (strict separation).}\ If the sets are a \textbf{positive distance} apart
  ($d=\inf\|a-b\|>0$; e.g.\ both compact and disjoint), the same $\mathbf{p}$ gives
  \textbf{strict} separation $\mathbf{p}\cdot a<\beta<\mathbf{p}\cdot b$ --- \AppendixLink{sec:separation-strict} (\InterestOnly).

  \medskip
  {\scriptsize Proof idea when $A$ is compact: \AppendixLink{sec:separation-proof} (\InterestOnly).}
\end{frame}

\begin{frame}{3.2 Example --- producer theory (numerical illustration)}
  \footnotesize
  \textbf{Netput plan} $y=(y_1,y_2)$:\ input $y_1\le 0$, output $y_2\ge 0$ (second quadrant).

  \medskip
  \textbf{Technology.}\ With $L=-y_1$ units of labor, max output is $f(L)=2\sqrt{L}$, so
  \[
    Y=\{(y_1,y_2):\ y_1\le 0,\ 0\le y_2\le 2\sqrt{-y_1}\}
    \quad\text{(free disposal on output).}
  \]

  \medskip
  \textbf{Prices.}\ Wage $p_1=1$, output price $p_2=2$.
  Profit $\pi=p_1y_1+p_2y_2$ (since $y_1<0$, the term $p_1y_1$ is labor cost).

  \medskip
  \textbf{Efficient plan.}\ Use $L=4$:\ $y^*=(-4,\,4)$.
  Then $\pi=1\cdot(-4)+2\cdot 4=4$.

  \medskip
  \textbf{Isoprofit line.}\ $y_1+2y_2=4$, or $y_2=2-\tfrac12 y_1$.
  At $y_1=0$:\ \textbf{$y_2$-axis intercept} $y_2=\pi/p_2=2$.
  Slope $-p_1/p_2=-\tfrac12$ matches the frontier at $y^*$ (supporting prices).
\end{frame}

\begin{frame}{3.2 Example --- producer theory}
  \footnotesize
  \textbf{Setup.}\ A firm's \textbf{production possibility set} $Y\subset\mathbb{R}^n$ is
  \textbf{convex} (and closed in full treatments).

  \medskip
  \textbf{Statement.}\ If $Y$ is convex, the theorem guarantees that every
  \textbf{technically efficient} production plan $y\in Y$ is also a
  \textbf{profit-maximizing} choice for some nonzero price vector $\mathbf{p}$:
  \[
    \mathbf{p}\cdot y \ge \mathbf{p}\cdot y'
    \quad\text{for all }y'\in Y.
  \]

  \medskip
  \textbf{Separation picture.}\ Let $B$ be outputs \textbf{strictly better than} $y$
  (more of some good, no less of any). If $y$ is efficient, $Y\cap B=\varnothing$.
  A separating hyperplane gives supporting prices $\mathbf{p}$ at $y$.

\begin{frame}{3.3 Vectors in $\mathbb{R}^n$}
  \footnotesize
  \textbf{Vector.}\
  An ordered list of $n$ real numbers. We write a \textbf{column vector} as
  \[
    \mathbf{x}=
    \begin{pmatrix}
      x_1\\ x_2\\ \vdots\\ x_n
    \end{pmatrix}
    \in\mathbb{R}^n.
  \]
  Its \textbf{transpose} $\mathbf{x}^{\mathsf T}=(x_1,\ldots,x_n)$ is a \textbf{row vector}.

  \medskip
  \textbf{Linear combination.}\
  Given vectors $\mathbf{v}_1,\ldots,\mathbf{v}_k\in\mathbb{R}^n$ and scalars $c_1,\ldots,c_k$,
  \[
    c_1\mathbf{v}_1+c_2\mathbf{v}_2+\cdots+c_k\mathbf{v}_k
  \]
  is a \textbf{linear combination} of the $\mathbf{v}_j$.

  \medskip
  \textbf{Span.}\
  All linear combinations of $\mathbf{v}_1,\ldots,\mathbf{v}_k$ form the \textbf{span} of those vectors.
\end{frame}

\begin{frame}{3.3 Matrix--vector product --- columns of $A$}
  \footnotesize
  \textbf{Setup.}\
  $A\in\mathbb{R}^{m\times n}$ has columns $\mathbf{a}_1,\ldots,\mathbf{a}_n\in\mathbb{R}^m$ and
  $\mathbf{x}\in\mathbb{R}^n$ is a column vector.

  \medskip
  \textbf{Column view.}\
  $A\mathbf{x}$ is a linear combination of the \textbf{columns} of $A$:
  \[
    A\mathbf{x}
    = x_1\mathbf{a}_1+x_2\mathbf{a}_2+\cdots+x_n\mathbf{a}_n
    \in\mathbb{R}^m.
  \]
  The weights are the entries of $\mathbf{x}$.

  \medskip
  \textbf{Example.}\
  $A=\begin{pmatrix}1&2\\ 3&4\\ 5&6\end{pmatrix}$,
  $\mathbf{x}=\begin{pmatrix}x_1\\ x_2\end{pmatrix}$:
  \[
    A\mathbf{x}
    = x_1\begin{pmatrix}1\\ 3\\ 5\end{pmatrix}
    + x_2\begin{pmatrix}2\\ 4\\ 6\end{pmatrix}.
  \]
\end{frame}

\begin{frame}{3.3 Row vector times matrix --- rows of $A$}
  \footnotesize
  \textbf{Setup.}\
  $A\in\mathbb{R}^{m\times n}$, $\mathbf{x}\in\mathbb{R}^n$ (column), so $A\mathbf{x}\in\mathbb{R}^m$.

  \medskip
  \textbf{Row view (transpose both sides).}\
  $(A\mathbf{x})^{\mathsf T}=\mathbf{x}^{\mathsf T}A^{\mathsf T}$:
  \[
    \mathbf{x}^{\mathsf T}A^{\mathsf T}
    = x_1\,\mathbf{a}_1^{\mathsf T}+x_2\,\mathbf{a}_2^{\mathsf T}+\cdots+x_n\,\mathbf{a}_n^{\mathsf T}
    \in\mathbb{R}^{1\times m}.
  \]
  Dimensions: $(1\times n)(n\times m)=(1\times m)$.
  \textbf{Note:}\ $\mathbf{x}^{\mathsf T}A$ does \textbf{not} match when $A\in\mathbb{R}^{m\times n}$ with $m\neq n$.

  \medskip
  \textbf{Example.}\
  From the previous slide,
  \[
    \mathbf{x}^{\mathsf T}=(x_1,x_2),
    \qquad
    A^{\mathsf T}=
    \begin{pmatrix}
      1&3&5\\
      2&4&6
    \end{pmatrix}.
  \]
  Then
  \[
    \mathbf{x}^{\mathsf T}A^{\mathsf T}
    = x_1(1,3,5)+x_2(2,4,6).
  \]
\end{frame}

\begin{frame}{3.3 Two views of the same product}
  \footnotesize
  \textbf{Same numbers, two interpretations.}\
  If $A\in\mathbb{R}^{m\times n}$ and $\mathbf{x}\in\mathbb{R}^n$:
  \begin{itemize}
    \item \textbf{Column view:}\ $A\mathbf{x}=\sum_{j=1}^n x_j\,\mathbf{a}_j\in\mathbb{R}^m$.
    \item \textbf{Entry view:}\ $(A\mathbf{x})_i=\mathbf{r}_i^{\mathsf T}A\mathbf{x}
      = \sum_{j=1}^n a_{ij}x_j$ (dot product of row $i$ of $A$ with $\mathbf{x}$),
      where $\mathbf{r}_i\in\mathbb{R}^m$ has $i$-th entry $1$ and all other entries $0$
      ($\mathbf{r}_i=\mathbf{e}_i$).
  \end{itemize}

  \medskip
  \textbf{Transpose link.}\
  $(A\mathbf{x})^{\mathsf T}=\mathbf{x}^{\mathsf T}A^{\mathsf T}$:
  a \textbf{row vector} on the left combines \textbf{rows of $A^{\mathsf T}$} (i.e.\ transposes of
  columns of $A$).
\end{frame}

\begin{frame}{3.3 Matrix multiplication $AB$ --- example}
  \footnotesize
  \textbf{Example.}\
  $A=\begin{pmatrix}1&2\\ 3&4\end{pmatrix}$,
  $B=\begin{pmatrix}5&6\\ 7&8\end{pmatrix}$.

  \medskip
  \textbf{Column view (both columns of $AB$).}\
  \[
    (AB)\mathbf{e}_1
    = A\begin{pmatrix}5\\ 7\end{pmatrix}
    = 5\begin{pmatrix}1\\ 3\end{pmatrix}+7\begin{pmatrix}2\\ 4\end{pmatrix}
    = \begin{pmatrix}19\\ 43\end{pmatrix},
  \]
  \[
    (AB)\mathbf{e}_2
    = A\begin{pmatrix}6\\ 8\end{pmatrix}
    = 6\begin{pmatrix}1\\ 3\end{pmatrix}+8\begin{pmatrix}2\\ 4\end{pmatrix}
    = \begin{pmatrix}22\\ 50\end{pmatrix}.
  \]
  So $AB=\begin{pmatrix}19&22\\ 43&50\end{pmatrix}$.

  \medskip
  \textbf{Row view (practice).}\
  Find rows $1$ and $2$ of $AB$ using
  $(1,2)B$ and $(3,4)B$.

  \medskip
  {\scriptsize \textbf{Note (skip in class).}\
  $(1,2)B=(19,22)$ and $(3,4)B=(43,50)$ --- same matrix as above.}
\end{frame}

\begin{frame}{3.3 Matrix multiplication $AB$}
  \footnotesize
  \textbf{Setup.}\
  $A\in\mathbb{R}^{m\times n}$, $B\in\mathbb{R}^{n\times p}$, so $AB\in\mathbb{R}^{m\times p}$.

  \medskip
  \textbf{Column view.}\
  Column $j$ of $AB$ is $A$ times column $j$ of $B$:
  \[
    (AB)\,\mathbf{e}_j = A(B\mathbf{e}_j)
    = A\mathbf{b}_j
    = b_{1j}\mathbf{a}_1+b_{2j}\mathbf{a}_2+\cdots+b_{nj}\mathbf{a}_n.
  \]
  Each column of $AB$ is a linear combination of \textbf{columns of $A$} (weights from column $j$ of
  $B$).

  \medskip
  \textbf{Row view.}\
  Row $i$ of $AB$ is row $i$ of $A$ times $B$:
  \[
    \mathbf{r}_i^{\mathsf T}(AB)
    = (\mathbf{r}_i^{\mathsf T}A)B
    = a_{i1}\mathbf{s}_1^{\mathsf T}+\cdots+a_{in}\mathbf{s}_n^{\mathsf T},
  \]
  where $\mathbf{r}_i\in\mathbb{R}^m$ has $i$-th entry $1$ and all other entries $0$
  ($\mathbf{r}_i=\mathbf{e}_i$), so $\mathbf{r}_i^{\mathsf T}$ selects row $i$; and
  $\mathbf{s}_k^{\mathsf T}$ is row $k$ of $B$: a linear combination of \textbf{rows of $B$}.
\end{frame}

\begin{frame}{3.3 Basis}
  \footnotesize
  \textbf{Idea.}\
  A \textbf{basis} of $\mathbb{R}^n$ is a small set of vectors from which every vector in
  $\mathbb{R}^n$ can be built as a \textbf{linear combination} in exactly one way.

  \medskip
  \textbf{Standard basis of $\mathbb{R}^n$.}\
  For $j=1,\ldots,n$, define $\mathbf{e}_j$ to have a $1$ in entry $j$ and $0$ elsewhere:
  \[
    \mathbf{e}_1=
    \begin{pmatrix}1\\ 0\\ \vdots\\ 0\end{pmatrix},
    \quad
    \mathbf{e}_2=
    \begin{pmatrix}0\\ 1\\ \vdots\\ 0\end{pmatrix},
    \quad\ldots,\quad
    \mathbf{e}_n=
    \begin{pmatrix}0\\ 0\\ \vdots\\ 1\end{pmatrix}.
  \]

  \medskip
  \textbf{Coordinates.}\
  Every $\mathbf{x}\in\mathbb{R}^n$ can be written uniquely as
  \[
    \mathbf{x}=x_1\mathbf{e}_1+x_2\mathbf{e}_2+\cdots+x_n\mathbf{e}_n,
  \]
  where $x_1,\ldots,x_n$ are the \textbf{coordinates} (entries) of $\mathbf{x}$.
\end{frame}

\begin{frame}{3.3 Basis --- example ($\mathbb{R}^2$)}
  \footnotesize
  \textbf{Key point.}\
  A basis is \textbf{not unique}, but for a \textbf{fixed} basis every vector has a
  \textbf{unique} representation as a linear combination.

  \medskip
  \textbf{Two bases of $\mathbb{R}^2$.}\
  Standard: $\mathbf{e}_1=(1,0)$, $\mathbf{e}_2=(0,1)$.\
  Alternative: $\mathbf{v}_1=(1,0)$, $\mathbf{v}_2=(1,1)$.

  \medskip
  \textbf{Same vector, two representations.}\
  For $\mathbf{b}=(2,3)$:
  \[
    \mathbf{b}=2\mathbf{e}_1+3\mathbf{e}_2
    \qquad\text{and}\qquad
    \mathbf{b}=(-1)\mathbf{v}_1+3\mathbf{v}_2.
  \]
  Different bases give different weights $x_1,\ldots,x_n$, but within one basis the weights are
  \textbf{unique}.

  \medskip
  \textbf{Requirement for a basis $\{\mathbf{a}_1,\ldots,\mathbf{a}_n\}$.}\
  Every $\mathbf{b}\in\mathbb{R}^n$ can be written as
  $\mathbf{b}=x_1\mathbf{a}_1+\cdots+x_n\mathbf{a}_n$ in \textbf{exactly one way}.
\end{frame}

\begin{frame}{3.3 Basis --- existence and uniqueness}
  \footnotesize
  \textbf{Requirement for a basis $\{\mathbf{a}_1,\ldots,\mathbf{a}_n\}$.}\
  Every $\mathbf{b}\in\mathbb{R}^n$ can be written as
  $\mathbf{b}=x_1\mathbf{a}_1+\cdots+x_n\mathbf{a}_n$ in \textbf{exactly one way}.

  \medskip
  \textbf{Two parts.}\
  \begin{itemize}
    \item \textbf{Existence:}\ every $\mathbf{b}\in\mathbb{R}^n$ can be written as some linear
      combination of $\mathbf{a}_1,\ldots,\mathbf{a}_n$.
    \item \textbf{Uniqueness:}\ there is \textbf{only one} choice of weights
      $x_1,\ldots,x_n$.
  \end{itemize}

  \medskip
  \textbf{For $A\mathbf{x}=\mathbf{b}$ with $A\in\mathbb{R}^{m\times n}$.}\
  Then $\mathbf{x}\in\mathbb{R}^n$, $\mathbf{b}\in\mathbb{R}^m$, and
  $A=[\mathbf{a}_1\ \cdots\ \mathbf{a}_n]$ with $\mathbf{a}_j\in\mathbb{R}^m$ gives
  \[
    A\mathbf{x}=\mathbf{b}
    \quad\Longleftrightarrow\quad
    \mathbf{b}=x_1\mathbf{a}_1+\cdots+x_n\mathbf{a}_n.
  \]
\end{frame}

\begin{frame}{3.3 Existence --- column space}
  \footnotesize
  \textbf{Setup.}\
  $A\in\mathbb{R}^{m\times n}$, $\mathbf{x}\in\mathbb{R}^n$, $\mathbf{b}\in\mathbb{R}^m$,
  $A=[\mathbf{a}_1\ \cdots\ \mathbf{a}_n]$ with $\mathbf{a}_j\in\mathbb{R}^m$.

  \medskip
  \textbf{Column space.}\
  \[
    \operatorname{Col}(A)=\operatorname{span}\{\mathbf{a}_1,\ldots,\mathbf{a}_n\}\subseteq\mathbb{R}^m.
  \]

  \medskip
  \textbf{Existence of a solution.}\
  $A\mathbf{x}=\mathbf{b}$ has \textbf{at least one} solution
  $\Longleftrightarrow$ $\mathbf{b}\in\operatorname{Col}(A)$.

  \medskip
  \textbf{Meaning.}\
  $\mathbf{b}\in\operatorname{Col}(A)$ means $\mathbf{b}=x_1\mathbf{a}_1+\cdots+x_n\mathbf{a}_n$ for
  some $\mathbf{x}$ --- the \textbf{existence} part of the basis requirement, but only for this one
  $\mathbf{b}\in\mathbb{R}^m$ (and only when $\operatorname{Col}(A)=\mathbb{R}^m$ does it hold for
  \textbf{all} $\mathbf{b}$).
\end{frame}

\begin{frame}{3.3 Uniqueness --- linearly independent columns}
  \footnotesize
  \textbf{Setup.}\
  $A\in\mathbb{R}^{m\times n}$, $\mathbf{x}\in\mathbb{R}^n$, $\mathbf{b}\in\mathbb{R}^m$.

  \medskip
  \textbf{Linearly independent.}\
  Vectors $\mathbf{v}_1,\ldots,\mathbf{v}_k$ are \textbf{linearly independent} if
  \[
    c_1\mathbf{v}_1+c_2\mathbf{v}_2+\cdots+c_k\mathbf{v}_k=\mathbf{0}
    \quad\Longrightarrow\quad
    c_1=c_2=\cdots=c_k=0.
  \]
  \textbf{Linearly dependent:}\ not independent --- some nontrivial combination equals $\mathbf{0}$.

  \medskip
  \textbf{Uniqueness of a solution.}\
  If $A\mathbf{x}=\mathbf{b}$ has a solution, it is \textbf{unique} iff the columns
  $\mathbf{a}_1,\ldots,\mathbf{a}_n$ are \textbf{linearly independent}.
\end{frame}

\begin{frame}{3.3 Linear independence $\Leftrightarrow$ unique solution}
  \footnotesize
  \textbf{Claim.}\
  Fix $\mathbf{b}\in\mathbb{R}^m$. If $A\mathbf{x}=\mathbf{b}$ has \textbf{at least one} solution, then
  it has a \textbf{unique} solution iff $\mathbf{a}_1,\ldots,\mathbf{a}_n$ are linearly independent.

  \medskip
  \textbf{($\Rightarrow$) Independent $\Rightarrow$ unique.}\
  If $A\mathbf{x}=\mathbf{b}$ and $A\mathbf{y}=\mathbf{b}$, then
  $A(\mathbf{x}-\mathbf{y})=\mathbf{0}$, i.e.\
  \[
    (x_1-y_1)\mathbf{a}_1+\cdots+(x_n-y_n)\mathbf{a}_n=\mathbf{0}.
  \]
  Independence gives $x_j-y_j=0$ for all $j$, so $\mathbf{x}=\mathbf{y}$.

  \medskip
  \textbf{($\Leftarrow$) Dependent $\Rightarrow$ not unique.}\
  If the columns are dependent, some $\mathbf{d}\neq\mathbf{0}$ satisfies
  $A\mathbf{d}=\mathbf{0}$.
  If $\mathbf{x}$ solves $A\mathbf{x}=\mathbf{b}$, then so does $\mathbf{x}+\mathbf{d}$:
  \[
    A(\mathbf{x}+\mathbf{d})=A\mathbf{x}+A\mathbf{d}=\mathbf{b}+\mathbf{0}=\mathbf{b},
  \]
  and $\mathbf{x}+\mathbf{d}\neq\mathbf{x}$. So the solution is \textbf{not} unique.
\end{frame}

\begin{frame}{3.3 Linear independence --- examples ($\mathbb{R}^2$)}
  \footnotesize
  \textbf{Independent (bases).}\
  \begin{itemize}
    \item Standard: $\mathbf{e}_1=(1,0)$, $\mathbf{e}_2=(0,1)$.
    \item Alternative: $\mathbf{v}_1=(1,0)$, $\mathbf{v}_2=(1,1)$.
  \end{itemize}
  For each pair, $c_1\mathbf{w}_1+c_2\mathbf{w}_2=\mathbf{0}$ implies $c_1=c_2=0$.

  \medskip
  \textbf{Dependent (too many vectors).}\
  $\mathbf{e}_1=(1,0)$, $\mathbf{e}_2=(0,1)$, and $\mathbf{u}=(1,1)$:
  \[
    \mathbf{u}=\mathbf{e}_1+\mathbf{e}_2
    \quad\Longleftrightarrow\quad
    \mathbf{e}_1+\mathbf{e}_2-\mathbf{u}=\mathbf{0}.
  \]
  Coefficients $(1,1,-1)$ are not all zero, so these three vectors are \textbf{linearly dependent}.
  A basis of $\mathbb{R}^2$ has exactly $2$ vectors, not $3$.
\end{frame}

\begin{frame}{3.3 Linear dependence --- columns example}
  \footnotesize
  \textbf{Example (columns of $A$).}\
  $A=\begin{pmatrix}1&2\\ 2&4\end{pmatrix}$, columns
  $\mathbf{a}_1=(1,2)^{\mathsf T}$, $\mathbf{a}_2=(2,4)^{\mathsf T}$.

  \medskip
  \textbf{Same direction.}\
  $\mathbf{a}_2=2\mathbf{a}_1$, so the two columns lie on one line through the origin.

  \medskip
  \textbf{Nontrivial combination to $\mathbf{0}$.}\
  \[
    2\mathbf{a}_1-\mathbf{a}_2=\mathbf{0}
    \quad\Longleftrightarrow\quad
    2\begin{pmatrix}1\\ 2\end{pmatrix}-\begin{pmatrix}2\\ 4\end{pmatrix}
    =\begin{pmatrix}0\\ 0\end{pmatrix}.
  \]
  Coefficients $(2,-1)$ are not both zero, so $\mathbf{a}_1,\mathbf{a}_2$ are
  \textbf{linearly dependent} (only one independent direction in $\operatorname{Col}(A)$).
\end{frame}

\begin{frame}{3.3 Solving $A\mathbf{x}=\mathbf{b}$ --- example}
  \footnotesize
  \textbf{Example.}\
  $A=\begin{pmatrix}1&2\\ 2&4\end{pmatrix}$, so
  $\operatorname{Col}(A)=\operatorname{span}\{(1,2)\}$ (a line in $\mathbb{R}^2$).

  \medskip
  \textbf{Solvable.}\
  $\mathbf{b}=\begin{pmatrix}3\\ 6\end{pmatrix}=3(1,2)$, so
  $A\mathbf{x}=\mathbf{b}$ has solutions (e.g.\ $\mathbf{x}=(1,1)^{\mathsf T}$; not unique).

  \medskip
  \textbf{Not solvable (existence fails).}\
  $\mathbf{b}=\begin{pmatrix}1\\ 0\end{pmatrix}\notin\operatorname{Col}(A)$, so
  $A\mathbf{x}=\mathbf{b}$ has \textbf{no} solution.

  \medskip
  \textbf{Dependent columns (uniqueness fails).}\
  $\mathbf{a}_2=2\mathbf{a}_1$, so the columns are \textbf{linearly dependent}.
  When a solution exists, it is not unique (e.g.\ both $\mathbf{x}=(3,0)^{\mathsf T}$ and
  $\mathbf{x}=(1,1)^{\mathsf T}$ give $\mathbf{b}=(3,6)^{\mathsf T}$).
\end{frame}

\begin{frame}{3.3 Full basis requirement --- only when $m=n$}
  \footnotesize
  \textbf{Goal.}\
  When does \textbf{every} $\mathbf{b}\in\mathbb{R}^m$ satisfy
  $\mathbf{b}=x_1\mathbf{a}_1+\cdots+x_n\mathbf{a}_n$ in \textbf{exactly one way}?

  \medskip
  \textbf{Claim.}\
  For $A\in\mathbb{R}^{m\times n}$, this happens \textbf{only if $m=n$}.

  \medskip
  \textbf{Proof sketch.}\
  \begin{itemize}
    \item \textbf{Uniqueness} $\Rightarrow$ columns independent.
      In $\mathbb{R}^m$ at most $m$ vectors can be independent, so $n\le m$.
    \item \textbf{Existence for all $\mathbf{b}$} $\Rightarrow$ $\operatorname{Col}(A)=\mathbb{R}^m$.
      With only $n$ columns, $\dim(\operatorname{Col}(A))\le n$, so $n\ge m$.
    \item Therefore $m=n$.
  \end{itemize}
  If $n<m$, columns cannot span $\mathbb{R}^m$; if $n>m$, $n$ columns in $\mathbb{R}^m$ cannot be
  independent.
\end{frame}

\begin{frame}{3.3 Square case --- basis $=$ existence $+$ uniqueness}
  \footnotesize
  \textbf{Now $A\in\mathbb{R}^{n\times n}$, $\mathbf{b}\in\mathbb{R}^n$.}\
  Requirement for a basis $\{\mathbf{a}_1,\ldots,\mathbf{a}_n\}$:
  every $\mathbf{b}\in\mathbb{R}^n$ is
  $\mathbf{b}=x_1\mathbf{a}_1+\cdots+x_n\mathbf{a}_n$ in \textbf{exactly one way}.

  \medskip
  \textbf{Equivalent to $A\mathbf{x}=\mathbf{b}$.}\
  \[
    A\mathbf{x}=\mathbf{b}
    \quad\Longleftrightarrow\quad
    \mathbf{b}=x_1\mathbf{a}_1+\cdots+x_n\mathbf{a}_n.
  \]

  \medskip
  \textbf{Columns form a basis of $\mathbb{R}^n$} $\Longleftrightarrow$
  existence and uniqueness for \textbf{every} $\mathbf{b}$ $\Longleftrightarrow$
  $A\mathbf{x}=\mathbf{b}$ has a \textbf{unique} solution for every $\mathbf{b}\in\mathbb{R}^n$.
\end{frame}

\begin{frame}{3.3 Invertible matrices --- equivalent conditions}
  \footnotesize
  \textbf{Let $A\in\mathbb{R}^{n\times n}$.}\
  The following are \textbf{equivalent}:

  \medskip
  \begin{enumerate}
    \item $A$ is \textbf{invertible}: $\exists\,A^{-1}$ with $AA^{-1}=A^{-1}A=I_n$.
    \item For every $\mathbf{b}\in\mathbb{R}^n$, $A\mathbf{x}=\mathbf{b}$ has a \textbf{unique}
      solution $\mathbf{x}=A^{-1}\mathbf{b}$.
    \item Columns $\mathbf{a}_1,\ldots,\mathbf{a}_n$ form a \textbf{basis} of $\mathbb{R}^n$.
    \item Columns are \textbf{linearly independent}
      ($\operatorname{Col}(A)=\mathbb{R}^n$).
    \item $A\mathbf{x}=\mathbf{0}$ has only the trivial solution $\mathbf{x}=\mathbf{0}$.
    \item Rows of $A$ are linearly independent (equivalently, form a basis of $\mathbb{R}^n$).
    \item $\det(A)\neq 0$.
  \end{enumerate}

  \medskip
  \textbf{Next:}\ what $\det(A)$ means geometrically, and why (6)--(7) fit this list.
\end{frame}

\begin{frame}{3.3 Determinant --- area scaling ($2\times 2$)}
  \footnotesize
  \textbf{Setup.}\
  $A=\begin{pmatrix}a&b\\ c&d\end{pmatrix}$, columns
  $\mathbf{a}_1=(a,c)^{\mathsf T}$, $\mathbf{a}_2=(b,d)^{\mathsf T}$.

  \medskip
  \textbf{Where the standard basis goes.}\
  \[
    A\begin{pmatrix}1\\ 0\end{pmatrix}=\begin{pmatrix}a\\ c\end{pmatrix}=\mathbf{a}_1,
    \qquad
    A\begin{pmatrix}0\\ 1\end{pmatrix}=\begin{pmatrix}b\\ d\end{pmatrix}=\mathbf{a}_2.
  \]
  So $(1,0)^{\mathsf T}$ is sent to $(a,c)^{\mathsf T}$, and $(0,1)^{\mathsf T}$ to $(b,d)^{\mathsf T}$.

  \medskip
  \textbf{Area conversion factor.}\
  The unit square (spanned by $(1,0)^{\mathsf T}$ and $(0,1)^{\mathsf T}$, area $=1$) is mapped by $A$ to
  the parallelogram spanned by $\mathbf{a}_1$ and $\mathbf{a}_2$.
  The \textbf{determinant} is exactly the (signed) area of that parallelogram:
  \[
    \det(A)=ad-bc.
  \]
  So $\det(A)$ is the \textbf{area conversion factor}: every region's area is multiplied by
  $|\det(A)|$ under $\mathbf{x}\mapsto A\mathbf{x}$.
\end{frame}

\begin{frame}{3.3 Determinant $=0$ --- example}
  \footnotesize
  \textbf{Example.}\
  $A=\begin{pmatrix}1&1\\ 1&1\end{pmatrix}$, so $a=b=c=d=1$ and
  \[
    \det(A)=ad-bc=1\cdot 1-1\cdot 1=0.
  \]

  \medskip
  \textbf{Standard basis collapses to a line.}\
  \[
    A\begin{pmatrix}1\\ 0\end{pmatrix}=\begin{pmatrix}1\\ 1\end{pmatrix},
    \qquad
    A\begin{pmatrix}0\\ 1\end{pmatrix}=\begin{pmatrix}1\\ 1\end{pmatrix}.
  \]
  Both $(1,0)^{\mathsf T}$ and $(0,1)^{\mathsf T}$ are sent to the \textbf{same} point on one line through
  the origin. The unit square (area $1$) is squashed to a \textbf{1-dimensional} set --- area $0$.

  \medskip
  \textbf{So (square case):}\
  \[
    \det(A)=0
    \quad\Longleftrightarrow\quad
    \text{columns linearly dependent}
    \quad\Longleftrightarrow\quad
    A\ \text{not invertible}.
  \]
  Here $\mathbf{a}_2=\mathbf{a}_1=(1,1)^{\mathsf T}$, so the columns are dependent and $A$ has no
  inverse.
\end{frame}

\begin{frame}{3.3 Two views of a matrix $A$}
  \footnotesize
  \textbf{View 1 --- $\mathbf{x}$ as weights.}\
  Write $A=[\mathbf{a}_1\ \cdots\ \mathbf{a}_n]$. Then
  \[
    A\mathbf{x}=x_1\mathbf{a}_1+x_2\mathbf{a}_2+\cdots+x_n\mathbf{a}_n.
  \]
  The entries of $\mathbf{x}$ are the \textbf{weights} on the columns of $A$.

  \medskip
  \textbf{View 2 --- $A$ as a linear transformation.}\
  Define $T:\mathbb{R}^n\to\mathbb{R}^m$ by $T(\mathbf{x})=A\mathbf{x}$.
  The matrix $A$ sends each input vector $\mathbf{x}$ to an output vector in $\mathbb{R}^m$;
  for square $A$, $|\det(A)|$ is the area-scaling factor (previous slides).
\end{frame}

\begin{frame}{3.3 Linear transformation --- definition}
  \footnotesize
  \textbf{Linear transformation.}\
  A map $T:\mathbb{R}^n\to\mathbb{R}^m$ is \textbf{linear} if for all $\mathbf{x},\mathbf{y}\in\mathbb{R}^n$
  and all scalars $c\in\mathbb{R}$:
  \[
    T(\mathbf{x}+\mathbf{y})=T(\mathbf{x})+T(\mathbf{y}),
    \qquad
    T(c\mathbf{x})=cT(\mathbf{x}).
  \]
  Equivalently, for all $\mathbf{x},\mathbf{y}$ and $c$:
  \[
    T(c\mathbf{x}+\mathbf{y})=cT(\mathbf{x})+T(\mathbf{y}).
  \]

  \medskip
  \textbf{Examples.}\
  Rotation, reflection, projection, and shear in the plane are linear.
  A map with a constant term, e.g.\ $T(\mathbf{x})=A\mathbf{x}+\mathbf{b}$ with $\mathbf{b}\neq\mathbf{0}$,
  is \textbf{not} linear.

  \medskip
  \textbf{Matrix form.}\
  Every linear $T$ is $T(\mathbf{x})=A\mathbf{x}$ for some matrix $A$ (general recipe on the next
  slides).
  For $A\in\mathbb{R}^{2\times 2}$, $|\det(A)|$ is the area-scaling factor from the determinant slides.
\end{frame}

\begin{frame}{3.3 Linear maps --- decompose, then combine}
  \footnotesize
  \textbf{Benefit.}\
  A linear map $T:\mathbb{R}^2\to\mathbb{R}^2$ may be hard to visualize on an \textbf{arbitrary}
  input $\mathbf{v}$.

  \medskip
  \textbf{But you often know $T$ on a few vectors.}\
  Suppose you already know $T(\mathbf{x})$ and $T(\mathbf{y})$, and $\mathbf{v}$ can be decomposed as
  \[
    \mathbf{v}=a\mathbf{x}+b\mathbf{y}
  \]
  for some scalars $a,b\in\mathbb{R}$.
  Then by linearity,
  \[
    T(\mathbf{v})
    = T(a\mathbf{x}+b\mathbf{y})
    = aT(\mathbf{x})+bT(\mathbf{y}).
  \]
  \textbf{Decompose} $\mathbf{v}$ into known pieces, \textbf{transform each piece}, then
  \textbf{combine}.

  \medskip
  \textbf{Standard basis recipe ($\mathbb{R}^2$).}\
  Take $\mathbf{x}=\mathbf{e}_1$, $\mathbf{y}=\mathbf{e}_2$, so $\mathbf{v}=v_1\mathbf{e}_1+v_2\mathbf{e}_2$.
  If you know $T(\mathbf{e}_1)$ and $T(\mathbf{e}_2)$, then
  $T(\mathbf{v})=v_1 T(\mathbf{e}_1)+v_2 T(\mathbf{e}_2)$.
  Stack $T(\mathbf{e}_1)$ and $T(\mathbf{e}_2)$ as columns to get the matrix of $T$ (next slide).
\end{frame}

\begin{frame}{3.3 Decompose, transform, combine --- notation}
  \footnotesize
  \textbf{Recipe.}\
  For $\mathbf{v}=v_1\mathbf{e}_1+v_2\mathbf{e}_2$ and linear $T$:

  \medskip
  \textbf{1.\ Decompose} the input:
  \[
    \mathbf{v}=v_1\mathbf{e}_1+v_2\mathbf{e}_2.
  \]

  \medskip
  \textbf{2.\ Transform} each piece (use known values of $T$ on $\mathbf{e}_1,\mathbf{e}_2$):
  \[
    v_1 T(\mathbf{e}_1),
    \qquad
    v_2 T(\mathbf{e}_2).
  \]

  \medskip
  \textbf{3.\ Combine} the outputs:
  \[
    T(\mathbf{v})
    =v_1 T(\mathbf{e}_1)+v_2 T(\mathbf{e}_2)
    =A\mathbf{v},
    \qquad
    A=\bigl[T(\mathbf{e}_1)\ \ T(\mathbf{e}_2)\bigr].
  \]
\end{frame}

\begin{frame}{3.3 Every linear map is a matrix}
  \footnotesize
  \textbf{Claim.}\
  If $T:\mathbb{R}^n\to\mathbb{R}^m$ is linear, then there is a unique matrix
  $A\in\mathbb{R}^{m\times n}$ such that $T(\mathbf{x})=A\mathbf{x}$ for all $\mathbf{x}$.

  \medskip
  \textbf{Construction.}\
  Let $\mathbf{e}_1,\ldots,\mathbf{e}_n$ be the standard basis of $\mathbb{R}^n$.
  Form $A$ by stacking the outputs as columns:
  \[
    A=\bigl[T(\mathbf{e}_1)\ \ T(\mathbf{e}_2)\ \ \cdots\ \ T(\mathbf{e}_n)\bigr].
  \]

  \medskip
  \textbf{Why it works.}\
  Any $\mathbf{x}=\sum_{j=1}^n x_j\mathbf{e}_j$. By linearity,
  \[
    T(\mathbf{x})
    =\sum_{j=1}^n x_j T(\mathbf{e}_j)
    =\sum_{j=1}^n x_j \mathbf{a}_j
    =A\mathbf{x}.
  \]
  Column $j$ of $A$ is exactly where $\mathbf{e}_j$ is sent.
\end{frame}

\begin{frame}{3.3 Null space --- directions squeezed to $\mathbf{0}$}
  \footnotesize
  \textbf{Setup ($A\in\mathbb{R}^{2\times 2}$).}\
  View $A$ as a linear transformation $T:\mathbb{R}^2\to\mathbb{R}^2$, $T(\mathbf{x})=A\mathbf{x}$.

  \medskip
  \textbf{Null space / kernel} (subspace of $\mathbb{R}^2$):
  \[
    \operatorname{Null}(A)=\ker(T)=\{\mathbf{x}\in\mathbb{R}^2:\ A\mathbf{x}=\mathbf{0}\}.
  \]
  Vectors in $\operatorname{Null}(A)$ are mapped to $\mathbf{0}$ in the output.

  \medskip
  \textbf{Intuition.}\
  If $\mathbf{x}\neq\mathbf{0}$ lies in $\operatorname{Null}(A)$, the whole line
  $\operatorname{span}\{\mathbf{x}\}\subset\mathbb{R}^2$ is sent to the single point
  $\mathbf{0}\in\mathbb{R}^2$ --- that \textbf{input direction is squeezed} (lost in the output).

  \medskip
  \textbf{Example ($\det A=0$).}\
  $A=\begin{pmatrix}1&1\\ 1&1\end{pmatrix}$:
  $A(1,-1)^{\mathsf T}=\mathbf{0}$, so
  $\operatorname{Null}(A)=\operatorname{span}\{(1,-1)^{\mathsf T}\}$ --- one direction squeezed.
\end{frame}

\begin{frame}{3.3 Rank--nullity --- squeezed vs.\ persist ($2\times 2$)}
  \footnotesize
  \textbf{Setup.}\
  $A\in\mathbb{R}^{2\times 2}$, $T:\mathbb{R}^2\to\mathbb{R}^2$, $T(\mathbf{x})=A\mathbf{x}$.
  The plane has $2$ input dimensions; under $T$, each splits into one of two fates:

  \medskip
  \textbf{1.\ Squeezed to $\mathbf{0}$} (kernel, in $\mathbb{R}^2$):\
  $\ker(T)=\operatorname{Null}(A)$; dimension $=\operatorname{nullity}(A)$.

  \medskip
  \textbf{2.\ Persist in the output} (image, in $\mathbb{R}^2$):\
  $\operatorname{Im}(T)=\operatorname{Col}(A)=\{A\mathbf{x}:\mathbf{x}\in\mathbb{R}^2\}$;
  dimension $=\operatorname{rank}(A)$ --- independent directions that \textbf{survive} in the output
  (not collapsed to $\mathbf{0}$).

  \medskip
  \textbf{Rank--nullity.}\
  $\operatorname{rank}(A)+\operatorname{nullity}(A)=2$.

  \medskip
  \textbf{Example ($\det A=0$).}\
  $A=\begin{pmatrix}1&1\\ 1&1\end{pmatrix}$:
  nullity $=1$ (direction $(1,-1)^{\mathsf T}$ squeezed),
  rank $=1$ (outputs fill the line $\operatorname{span}\{(1,1)^{\mathsf T}\}$).

  \medskip
  \textbf{Example ($\det A\neq 0$).}\
  Shear or rotation from earlier slides:
  nullity $=0$ (only $\mathbf{0}$ squeezed),
  rank $=2$ (both input dimensions persist --- output still fills $\mathbb{R}^2$).
\end{frame}

\begin{frame}{3.3 Rank--nullity --- general case (\InterestOnly)}
  \footnotesize
  \textbf{Setup.}\
  Let $A\in\mathbb{R}^{m\times n}$, $T:\mathbb{R}^n\to\mathbb{R}^m$, $T(\mathbf{x})=A\mathbf{x}$.

  \medskip
  \textbf{Kernel and image.}\
  $\ker(T)=\operatorname{Null}(A)=\{\mathbf{x}\in\mathbb{R}^n:A\mathbf{x}=\mathbf{0}\}\subseteq\mathbb{R}^n$;
  \quad
  $\operatorname{Im}(T)=\operatorname{Col}(A)=\{A\mathbf{x}:\mathbf{x}\in\mathbb{R}^n\}\subseteq\mathbb{R}^m$.

  \medskip
  \textbf{Rank and nullity.}\
  $\operatorname{rank}(A)=\dim(\operatorname{Im}\,T)$,
  \quad
  $\operatorname{nullity}(A)=\dim(\ker T)$.

  \medskip
  \textbf{Rank--nullity theorem.}\
  \[
    \operatorname{rank}(A)+\operatorname{nullity}(A)=n
    \qquad\text{($n$ columns $=\dim(\mathbb{R}^n)$).}
  \]

  \medskip
  \textbf{Same intuition.}\
  Of the $n$ domain dimensions, $\operatorname{nullity}(A)$ are \textbf{squeezed} to
  $\mathbf{0}\in\mathbb{R}^m$; $\operatorname{rank}(A)$ independent directions
  \textbf{persist} in $\operatorname{Im}(T)\subseteq\mathbb{R}^m$.
  When $n=m$, invertible $\Leftrightarrow$ nullity $=0$ $\Leftrightarrow$ rank $=n$.
\end{frame}

\begin{frame}{3.3 After a squeeze --- what survives? ($2\times 2$)}
  \footnotesize
  \textbf{Recall.}\
  $A=\begin{pmatrix}1&1\\ 1&1\end{pmatrix}$ maps the whole plane onto one output line
  (through $(1,1)^{\mathsf T}$). Direction $(1,-1)^{\mathsf T}$ is squeezed to $\mathbf{0}$.

  \medskip
  \textbf{Question.}\
  Given an input $\mathbf{v}$, what is $A\mathbf{v}$?

  \medskip
  \textbf{Key idea (in words).}\
  Think of $\mathbf{v}$ as two moves: one along the \textbf{squeezed} direction (the map
  \textbf{forgets} it) and one along the surviving direction $(1,1)^{\mathsf T}$ (the map
  \textbf{remembers} only this). So $A\mathbf{v}$ depends on how much of $\mathbf{v}$ points
  along the surviving direction --- not on the part that collapses.

  \medskip
  \textbf{Example.}\
  $(3,1)^{\mathsf T}=2(1,1)^{\mathsf T}+1(1,-1)^{\mathsf T}$: the $(1,-1)^{\mathsf T}$ part is
  lost, the $2(1,1)^{\mathsf T}$ part survives, so $A\mathbf{v}=2(2,2)^{\mathsf T}=(4,4)^{\mathsf T}$.
\end{frame}

\begin{frame}{3.3 Dot product --- algebraic and geometric}
  \footnotesize
  For $\mathbf{u},\mathbf{v}\in\mathbb{R}^n$:

  \medskip
  \textbf{Algebraic definition.}
  \[
    \mathbf{u}\cdot\mathbf{v}=\sum_{i=1}^n u_i v_i.
  \]

  \medskip
  \textbf{Geometric definition.}\
  Let $\theta$ be the angle between $\mathbf{u}$ and $\mathbf{v}$. Then
  \[
    \mathbf{u}\cdot\mathbf{v}=\|\mathbf{u}\|\,\|\mathbf{v}\|\cos\theta.
  \]

  \medskip
  \textbf{Equivalent.}\
  These two definitions always give the \textbf{same number}.

  \medskip
  \textbf{Video (intuition and proof).}\
  \href{https://www.youtube.com/watch?v=LyGKycYT2v0}{\textcolor{mAccent}{https://www.youtube.com/watch?v=LyGKycYT2v0}}
\end{frame}

\begin{frame}{3.3 Linear regression --- $\mathbf{y}$ and $X$}
  \footnotesize
  \textbf{Data ($n$ workers).}\
  Worker $i$ has income $y_i$, age $\text{age}_i$, education $\text{educ}_i$.

  \medskip
  \textbf{Response vector.}
  \[
    \mathbf{y}=
    \begin{pmatrix} y_1 \\ \vdots \\ y_n \end{pmatrix}
    \in\mathbb{R}^n.
  \]

  \medskip
  \textbf{Regressor matrix} (two columns: age, educ).
  \[
    X=
    \begin{pmatrix}
      \text{age}_1 & \text{educ}_1 \\
      \vdots & \vdots \\
      \text{age}_n & \text{educ}_n
    \end{pmatrix}
    \in\mathbb{R}^{n\times 2},
    \qquad
    \boldsymbol{\beta}=
    \begin{pmatrix}\beta_{\text{age}}\\ \beta_{\text{educ}}\end{pmatrix},
    \qquad
    \mathbf{y}\approx X\boldsymbol{\beta}.
  \]
  Column $j$ of $X$ is the vector of worker $i$ values for regressor $j$.

  \medskip
  {\scriptsize Schematic next ($n=3$): age and education are the two axes; yellow plane $=\operatorname{Col}(X)$;
  $\mathbf{e}$ is orthogonal to every column of $X$.}
\end{frame}

\begin{frame}{3.3 OLS fit --- $\mathbf{e}$ orthogonal to every column of $X$}
  \footnotesize
  \textbf{From the picture ($n=3$).}\
  Yellow plane $=\operatorname{Col}(X)$; fitted $\widehat{\mathbf{y}}=X\widehat{\boldsymbol{\beta}}$ lies in it.
  Residual $\mathbf{e}=\mathbf{y}-X\widehat{\boldsymbol{\beta}}$ is perpendicular to the plane.

  \medskip
  \textbf{Orthogonal to every column of $X$.}\
  Let $\mathbf{x}_1$ = age column, $\mathbf{x}_2$ = educ column. Then
  \[
    \mathbf{x}_1\cdot\mathbf{e}=0,
    \qquad
    \mathbf{x}_2\cdot\mathbf{e}=0
    \qquad\Longleftrightarrow\qquad
    X^{\mathsf T}\mathbf{e}=\mathbf{0}
    \qquad\Longleftrightarrow\qquad
    X^{\mathsf T}\bigl(\mathbf{y}-X\widehat{\boldsymbol{\beta}}\bigr)=\mathbf{0}.
  \]
  These \textbf{normal equations} characterise $\widehat{\boldsymbol{\beta}}$.

  \medskip
  \textbf{Solve for $\widehat{\boldsymbol{\beta}}$.}\
  $X^{\mathsf T}X\widehat{\boldsymbol{\beta}}=X^{\mathsf T}\mathbf{y}$. If the columns of $X$ are
  independent (full column rank), then
  \[
    \widehat{\boldsymbol{\beta}}=(X^{\mathsf T}X)^{-1}X^{\mathsf T}\mathbf{y}.
  \]
  {\scriptsize (Since $\widehat{\mathbf{y}}\in\operatorname{Col}(X)$, also $\widehat{\mathbf{y}}\cdot\mathbf{e}=0$.)}
\end{frame}

\begin{frame}{3.3 OLS fit --- same as least squares}
  \footnotesize
  \textbf{Pythagoras in $\mathbb{R}^n$.}\
  Fix $\widehat{\boldsymbol{\beta}}$ with $\mathbf{e}=\mathbf{y}-X\widehat{\boldsymbol{\beta}}$. For any $\boldsymbol{\beta}$,
  \[
    \mathbf{y}-X\boldsymbol{\beta}
    =\underbrace{\mathbf{y}-X\widehat{\boldsymbol{\beta}}}_{\mathbf{e}}
    +\underbrace{X(\widehat{\boldsymbol{\beta}}-\boldsymbol{\beta})}_{\in\,\operatorname{Col}(X)}.
  \]
  The two pieces are orthogonal, so
  \[
    \|\mathbf{y}-X\boldsymbol{\beta}\|^2
    =\|\mathbf{y}-X\widehat{\boldsymbol{\beta}}\|^2+\|X(\widehat{\boldsymbol{\beta}}-\boldsymbol{\beta})\|^2
    \ge \|\mathbf{y}-X\widehat{\boldsymbol{\beta}}\|^2.
  \]

  \medskip
  \textbf{Conclusion.}\
  The smallest squared error occurs when $X(\widehat{\boldsymbol{\beta}}-\boldsymbol{\beta})=\mathbf{0}$, i.e.\
  $\widehat{\boldsymbol{\beta}}=\arg\min_{\boldsymbol{\beta}}\|\mathbf{y}-X\boldsymbol{\beta}\|^2
  =\arg\min_{\boldsymbol{\beta}}\sum_{i=1}^n
  \bigl(y_i-\beta_{\text{age}}\,\text{age}_i-\beta_{\text{educ}}\,\text{educ}_i\bigr)^2$.
\end{frame}

\begin{frame}{3.3 Which basis makes decompose--transform--combine easy?}
  \footnotesize
  \textbf{Recall the recipe.}\
  If $\mathbf{v}=a\mathbf{x}+b\mathbf{y}$, then
  $T(\mathbf{v})=aT(\mathbf{x})+bT(\mathbf{y})$ --- always true by linearity.

  \medskip
  \textbf{But step 2 can be hard.}\
  You must know $T(\mathbf{x})$ and $T(\mathbf{y})$. With the standard basis
  ($\mathbf{e}_1,\mathbf{e}_2$), rotation sent $\mathbf{e}_1$ down and $\mathbf{e}_2$ right ---
  both outputs are \textbf{new directions} to track.

  \medskip
  \textbf{A proper basis (for this map).}\
  Pick directions where $T$ \textbf{only scales}, not rotates or shears:
  \[
    T(\mathbf{v})=\lambda\mathbf{v}
    \qquad\text{for some scalar $\lambda$.}
  \]
  Then $T(a\mathbf{v})=a\lambda\mathbf{v}$ --- one number $\lambda$, same direction.\
  These special directions are \textbf{eigenvectors}; $\lambda$ is the \textbf{eigenvalue}.
\end{frame}

\begin{frame}{3.3 Eigenvectors and eigenvalues --- definition}
  \footnotesize
  \textbf{Setup.}\
  Square matrix $A\in\mathbb{R}^{n\times n}$ (linear map $T(\mathbf{x})=A\mathbf{x}$).

  \medskip
  \textbf{Eigenvector / eigenvalue.}\
  A nonzero vector $\mathbf{v}$ is an \textbf{eigenvector} of $A$ if
  \[
    A\mathbf{v}=\lambda\mathbf{v}
  \]
  for some scalar $\lambda$ (the \textbf{eigenvalue}).\
  Geometrically: $A$ stretches or flips $\mathbf{v}$, but keeps its line.

  \medskip
  \textbf{Proper basis.}\
  When $A$ has enough independent eigenvectors
  $\mathbf{v}_1,\ldots,\mathbf{v}_n$, decompose in \textbf{that} basis instead of the standard one.
\end{frame}

\begin{frame}{3.3 Eigenvectors --- $2\times 2$ example}
  \footnotesize
  \textbf{Map (not diagonal).}\
  $A=\begin{pmatrix}2&1\\ 1&2\end{pmatrix}$ mixes $x_1$ and $x_2$; standard $\mathbf{e}_1,\mathbf{e}_2$ are
  \textbf{not} eigenvectors.

  \medskip
  \textbf{Characteristic equation.}\
  $\det(A-\lambda I)=0$:
  \[
    (2-\lambda)^2-1=\lambda^2-4\lambda+3=(\lambda-1)(\lambda-3)=0
    \quad\Rightarrow\quad \lambda_1=3,\ \lambda_2=1.
  \]

  \medskip
  \textbf{Eigenpairs.}\
  $\lambda_1=3$: $\mathbf{v}_1=\begin{pmatrix}1\\ 1\end{pmatrix}$;
  \quad
  $\lambda_2=1$: $\mathbf{v}_2=\begin{pmatrix}1\\ -1\end{pmatrix}$
  (check: $A\mathbf{v}_i=\lambda_i\mathbf{v}_i$).

  \medskip
  \textbf{Decompose--transform--combine.}\
  $\mathbf{v}=\begin{pmatrix}3\\ 1\end{pmatrix}=2\mathbf{v}_1+1\mathbf{v}_2$:
  \[
    A\mathbf{v}=2\cdot 3\mathbf{v}_1+1\cdot 1\mathbf{v}_2=\begin{pmatrix}7\\ 5\end{pmatrix}.
  \]
  Same directions; coefficients $2,1$ become $6,1$.
\end{frame}

\begin{frame}{3.3 Eigenbasis --- especially useful in high dimension}
  \footnotesize
  \textbf{In $\mathbb{R}^n$ with eigenbasis $\mathbf{v}_1,\ldots,\mathbf{v}_n$.}\
  Any $\mathbf{v}=\sum_{i=1}^n c_i\mathbf{v}_i$. By linearity,
  \[
    A\mathbf{v}
    =\sum_{i=1}^n c_i A\mathbf{v}_i
    =\sum_{i=1}^n c_i\lambda_i\mathbf{v}_i.
  \]
  Transform each piece independently: multiply coefficient $c_i$ by $\lambda_i$.

  \medskip
  \textbf{Why this matters when $n$ is large.}\
  Full matrix--vector multiply mixes all $n$ coordinates at once.
  In an eigenbasis, $A$ acts as \textbf{separate scalings} along $n$ directions ---
  much easier to describe and compute (e.g.\ powers $A^k\mathbf{v}=\sum_i c_i\lambda_i^k\mathbf{v}_i$).

  \medskip
  \textbf{Bottom line.}\
  When a full eigenbasis exists, decompose--transform--combine is as simple as possible ---
  and the same idea drives many tools in econometrics and macro (optional applications on the next slides).

  \medskip
  {\scriptsize Not every matrix has a full real eigenbasis (e.g.\ rotation in the plane);
  when it does not, related decompositions (SVD, QR) still play the same role.}
\end{frame}

\begin{frame}{3.3 Applications --- PCA and Markov chains \InterestOnly}
  \footnotesize
  {\scriptsize \InterestOnly\
  You may need to understand some of these when you pursue macro, empirical economics, or IO.}

  \medskip
  \textbf{Extremely useful beyond the $2\times 2$ example.}\
  Many applied problems reduce to: find directions where a matrix \textbf{acts simply}.

  \medskip
  \begin{itemize}
    \item \textbf{PCA (principal components).}\
      Eigenvectors of the \textbf{covariance matrix} are the main directions of variation in data;
      largest eigenvalues pick the most important components for dimension reduction.
    \item \textbf{Markov transition matrices.}\
      For a stochastic matrix $P$ (rows sum to $1$), the \textbf{long-run distribution} is an eigenvector
      with $\lambda=1$;
      other eigenvalues control how fast $P^k\mathbf{v}$ converges to that steady state.
  \end{itemize}

  \medskip
  \textbf{Pattern.}\
  Decompose a state or data vector in special directions; each direction evolves by a single number ($\lambda_i$).
\end{frame}

\begin{frame}{3.3 Applications --- ODEs \InterestOnly}
  \footnotesize
  \textbf{Ordinary differential equations (ODEs) --- eigenvalues / eigenvectors show up in:}
  \begin{itemize}
    \item Linear systems $\dot{\mathbf{x}}=A\mathbf{x}$
    \item Stability of equilibria and phase-plane dynamics
    \item Continuous-time macro and growth models
    \item Linearized dynamics around steady states
    \item Compartment models (e.g.\ epidemiology, population dynamics)
  \end{itemize}
\end{frame}

\begin{frame}{3.3 Decompositions --- QR, eigen, SVD \InterestOnly}
  \footnotesize
  \textbf{Different decompositions, same spirit:}\
  rewrite a matrix so that the hard part is \textbf{diagonal}, \textbf{triangular}, or \textbf{orthogonal}.

  \medskip
  \begin{itemize}
    \item \textbf{QR decomposition} ($A=QR$).\
      $Q$ has orthonormal columns spanning $\mathrm{Col}(A)$;
      $R$ is upper triangular.
      Used to pick \textbf{linearly independent columns}, solve least squares stably, and build regression numerically.
    \item \textbf{Spectral / eigen decomposition} ($A=V\Lambda V^{-1}$).\
      When $A$ has a full eigenbasis, $V$ lists eigenvectors and $\Lambda$ lists eigenvalues ---
      matrix powers and dynamics become coefficient-wise.
    \item \textbf{Singular Value Decomposition (SVD)} ($A=U\Sigma V^{\mathsf T}$).\
      Always exists;
      built from eigenproblems on $A^{\mathsf T}A$ (and $AA^{\mathsf T}$).
      Underlies PCA, low-rank approximation, and regularized regression.
  \end{itemize}

\begin{frame}{A.1 Proof.\ Separation when $A$ is compact}
  \hypertarget{sec:separation-proof}{}
  \footnotesize
  \InterestOnly

  \medskip
  \textbf{Claim.}\ Disjoint nonempty convex $A,B\subset\mathbb{R}^n$ with $A$ compact and $B$ closed.
  Then $\exists\,\mathbf{p}\neq\mathbf{0}$, $\beta\in\mathbb{R}$ with
  $\mathbf{p}\cdot a\le\beta\le\mathbf{p}\cdot b$ for all $a\in A$, $b\in B$.

  \medskip
  \textbf{Step 1 --- closest pair.}\
  Define $d=\inf\{\|a-b\|:a\in A,\,b\in B\}>0$.
  Pick sequences $(a_n)\subset A$, $(b_n)\subset B$ with $\|a_n-b_n\|\to d$.
  Since $A$ is compact, a subsequence $a_{n_k}\to a_0\in A$.
  The $(b_{n_k})$ are bounded, so (using $B$ closed) a further subsequence $b_{n_{k_j}}\to b_0\in B$.
  Hence the infimum is \textbf{attained} at some pair $(a_0,b_0)\in A\times B$.

  \medskip
  {\scriptsize \textbf{Not an interior point.}\
  $a_0$ and $b_0$ need only lie in $A$ and $B$; they are usually on the \textbf{boundaries},
  not in $\operatorname{int}(A)$ or $\operatorname{int}(B)$.}

  \medskip
  \textbf{Step 2 --- normal vector.}\
  Set $\mathbf{p}=b_0-a_0\neq\mathbf{0}$ (points from $a_0$ toward $b_0$).
\end{frame}

\begin{frame}{A.1 Step 1 --- why the subsequence argument works}
  \footnotesize
  \InterestOnly

  \medskip
  \textbf{Why $a_{n_k}\to a_0\in A$.}\
  In $\mathbb{R}^n$, \textbf{compact} $=$ closed $+$ bounded (Heine--Borel).
  Every sequence in a compact set has a \textbf{subsequence} converging to a limit \textbf{in that set}.
  So $(a_n)\subset A$ compact $\Rightarrow\ \exists\,a_{n_k}\to a_0\in A$.

  \medskip
  \textbf{Why $(b_{n_k})$ is bounded.}\
  Since $\|a_{n_k}-b_{n_k}\|\to d$ and $(a_{n_k})$ converges, for large $k$,
  \[
    \|b_{n_k}\|
    \le \|b_{n_k}-a_{n_k}\|+\|a_{n_k}\|
    \le d+1+\|a_0\|+1.
  \]
  (Triangle inequality: $\|b\|\le\|b-a\|+\|a\|$.)

  \medskip
  \textbf{Why $b_{n_{k_j}}\to b_0\in B$.}\
  A bounded sequence in $\mathbb{R}^n$ has a convergent subsequence (Bolzano--Weierstrass).
  Since $B$ is \textbf{closed}, the limit $b_0$ lies in $B$.
\end{frame}

\begin{frame}{A.1 Step 3 --- why the derivative works}
  \footnotesize
  \InterestOnly

  \medskip
  \textbf{Setup.}\
  Fix $a\in A$. For $t\in[0,1]$, convexity gives $a_t=a_0+t(a-a_0)\in A$ and
  $g(t)=\|a_t-b_0\|^2$.
  Since $(a_0,b_0)$ minimizes distance, $t=0$ minimizes $g$ on $[0,1]$.

  \medskip
  \textbf{Why use the derivative?}\
  A minimum at the left endpoint $t=0$ means the slope cannot point downward:
  $g'(0)\ge 0$.

  \medskip
  \textbf{Differentiate.}\
  $g(t)=\|a_0+t(a-a_0)-b_0\|^2$, so
  \[
    g'(t)=2\langle a_0+t(a-a_0)-b_0,\,a-a_0\rangle,
    \qquad
    g'(0)=2\langle a_0-b_0,\,a-a_0\rangle\ge 0.
  \]
  Hence $\langle b_0-a_0,\,a-a_0\rangle\le 0$.
  With $\mathbf{p}=b_0-a_0$: $\mathbf{p}\cdot(a-a_0)\le 0$, so $\mathbf{p}\cdot a\le\mathbf{p}\cdot a_0$.
\end{frame}

\begin{frame}{A.1 Proof.\ Separation when $A$ is compact (continued)}
  \footnotesize
  \textbf{Step 4 --- $B$ stays on the other side.}\
  Fix $b\in B$. For $t\in[0,1]$, set $b_t=b_0+t(b-b_0)\in B$ and
  $h(t)=\|a_0-b_t\|^2$.
  Since $t=0$ minimizes $h$ on $[0,1]$, the same derivative argument gives
  $\mathbf{p}\cdot b\ge\mathbf{p}\cdot b_0$ for all $b\in B$.

  \medskip
  \textbf{Step 5 --- finish.}\
  Since $A\cap B=\varnothing$, the closest points satisfy $a_0\neq b_0$.
  With $\mathbf{p}=b_0-a_0$,
  \[
    \mathbf{p}\cdot b_0-\mathbf{p}\cdot a_0
    =\mathbf{p}\cdot(b_0-a_0)
    =\|b_0-a_0\|^2=d^2>0,
  \]
  so $\mathbf{p}\cdot a_0<\mathbf{p}\cdot b_0$.
  Any $\beta$ with $\mathbf{p}\cdot a_0\le\beta\le\mathbf{p}\cdot b_0$ works
  (e.g.\ $\beta=\mathbf{p}\cdot a_0$ gives \textbf{weak} separation:
  $\mathbf{p}\cdot a\le\beta\le\mathbf{p}\cdot b$ for all $a\in A$, $b\in B$).
\end{frame}

\begin{frame}{A.1 Strict separation}
  \hypertarget{sec:separation-strict}{}
  \footnotesize
  \InterestOnly

  \medskip
  \textbf{Weak vs.\ strict.}\
  From Steps~3--5: $\mathbf{p}\cdot a\le\mathbf{p}\cdot a_0$ and $\mathbf{p}\cdot b\ge\mathbf{p}\cdot b_0$.
  So any $\beta$ with $\mathbf{p}\cdot a_0\le\beta\le\mathbf{p}\cdot b_0$ separates weakly
  (e.g.\ $\beta=\mathbf{p}\cdot a_0$).

  \medskip
  \textbf{When is separation strict?}\
  If $d=\|b_0-a_0\|>0$ (sets bounded apart; e.g.\ both compact and disjoint), then
  $\mathbf{p}\cdot a_0<\mathbf{p}\cdot b_0$ with a \textbf{positive gap}.
  Pick $\beta$ \textbf{strictly between} them, e.g.\
  $\beta=\tfrac12(\mathbf{p}\cdot a_0+\mathbf{p}\cdot b_0)$. Then
  \[
    \mathbf{p}\cdot a\le\mathbf{p}\cdot a_0<\beta<\mathbf{p}\cdot b_0\le\mathbf{p}\cdot b
    \quad\Rightarrow\quad
    \mathbf{p}\cdot a<\beta<\mathbf{p}\cdot b.
  \]

  \medskip
  {\scriptsize If $d=0$ (sets touch in the limit, e.g.\ $A=[0,1]$, $B=(1,2]$), only weak separation is possible.}
\end{frame}

\begin{frame}{A.1 Proof.\ Remarks}
  \footnotesize
  \InterestOnly

  \medskip
  \textbf{If only $B$ is compact.}\
  Swap the roles of $A$ and $B$, apply the proof, then relabel the normal vector.

  \medskip
  \textbf{Single point $B=\{b_0\}$.}\
  Let $\hat a$ be a point in $A$ closest to $b_0$ (exists since $A$ is closed and convex).
  The same calculation with $(a_0,b_0)=(\hat a,b_0)$ separates $A$ from $b_0$ even when $A$ is not compact.
\end{frame}

\section{Producer theory example --- proof}

\begin{frame}{A.2 Proof.\ Efficient production $\Rightarrow$ profit maximization}
  \hypertarget{sec:producer-proof}{}
  \footnotesize
  \InterestOnly

  \medskip
  \textbf{Goal.}\ $y\in Y$ technically efficient $\Rightarrow\ \exists\,\mathbf{p}\neq\mathbf{0}$ with
  $\mathbf{p}\cdot y\ge\mathbf{p}\cdot y'$ for all $y'\in Y$.

  \medskip
  \textbf{Assumptions (camp level).}\
  $Y\subset\mathbb{R}^n$ is \textbf{closed and convex}; $Y$ is \textbf{bounded} (so compact);
  \textbf{free disposal} (if $y\in Y$ and $\tilde y\le y$, then $\tilde y\in Y$).

  \medskip
  \textbf{Step 1 --- better set.}\
  Define
  \[
    P_y=\{y'\in\mathbb{R}^n:\ y'\ge y,\ y'\neq y\}
  \]
  (weakly more output, strictly more in at least one good). $P_y$ is \textbf{convex}.

  \medskip
  \textbf{Step 2 --- disjointness.}\
  If $y'\in P_y\cap Y$, then $y'$ weakly dominates $y$ in $Y$, contradicting efficiency.
  So $P_y\cap Y=\varnothing$.
\end{frame}

\begin{frame}{A.2 Proof.\ Efficient production (continued)}
  \footnotesize
  \InterestOnly

  \medskip
  \textbf{Step 3 --- separate.}\
  $Y$ is compact and $P_y$ is closed, so apply the separating hyperplane theorem
  (swap labels if needed so a compact set is on one side).
  There exist $\mathbf{p}\neq\mathbf{0}$ and $\beta$ with
  \[
    \mathbf{p}\cdot y''\le\beta\le\mathbf{p}\cdot y'
    \quad\forall\,y''\in Y,\;\forall\,y'\in P_y.
  \]
  Hence $\mathbf{p}\cdot y''\le\mathbf{p}\cdot y'$ for all $y''\in Y$, $y'\in P_y$.

  \medskip
  \textbf{Step 4 --- profit maximization at $y$.}\
  For each good $i$, pick $y'^{(i)}=y+\varepsilon e_i\in P_y$ for small $\varepsilon>0$
  (feasible with free disposal). Then
  $\mathbf{p}\cdot y''\le\mathbf{p}\cdot y'^{(i)}=\mathbf{p}\cdot y+\varepsilon p_i$.
  Let $\varepsilon\downarrow 0$ to get $\mathbf{p}\cdot y''\le\mathbf{p}\cdot y$ for all $y''\in Y$.

  \medskip
  \textbf{Remark.}\
  Separation gives only $\mathbf{p}\neq\mathbf{0}$.
  For $\mathbf{p}\ge 0$, need free disposal on all goods: if $p_i<0$, then
  $y-\varepsilon e_i\in Y$ (since $\le y$) raises profit, contradicting maximization at $y$.
\end{frame}
